645 template <typename targetidentifierT>
646 void SynapticSamplingRewardGradientConnection<targetidentifierT>::send(nest::Event& e,
647  nest::thread thread,
648  double t_last_spike,
649  const CommonPropertiesType &cp)
650 {
651  if (is_degenerated())
652  {
653  // synapse is waiting for the garbage collector.
654  return;
655  }
656 
657  assert(cp.resolution_unit_ > 0.0);
658 
659  const long s_to = std::floor( e.get_stamp().get_ms() / cp.resolution_unit_ );
660  long s_from = std::floor( t_last_spike / cp.resolution_unit_ );
661 
662  if (s_to > s_from)
663  {
664  if (s_from == 0)
665  {
666  update_synapic_weight(0, cp);
667  }
668  // prepare the pointer to the target neuron. We can safely static_cast
669  // since the connection is checked when established.
670  TracingNode* target = static_cast<TracingNode*> (get_target(thread));
671 
672  TracingNode::const_iterator bap_trace =
673  target->get_trace(s_from, cp.bap_trace_id_);
674 
675  TracingNode::const_iterator dopa_trace =
676  cp.reward_transmitter_->get_trace(s_from, cp.dopa_trace_id_);
677 
678  const double t_last_weight_update =
679  std::floor(t_last_spike / cp.weight_update_interval_) * cp.weight_update_interval_;
680  const long s_last_update = std::floor( t_last_weight_update/cp.resolution_unit_ );
681 
682  for (long next_weight_step = s_last_update + cp.weight_update_steps_;
683  next_weight_step <= s_to;
684  next_weight_step += cp.weight_update_steps_)
685  {
686  update_synapse_state(next_weight_step, s_from, bap_trace, dopa_trace, cp);
687  update_synapic_parameter(thread, cp);
688  update_synapic_weight(next_weight_step, cp);
689  s_from = next_weight_step;
690  }
691 
692  if (s_to > s_from)
693  {
694  update_synapse_state(s_to, s_from, bap_trace, dopa_trace, cp);
695  }
696  }
697 
698  if (cp.delete_retracted_synapses_ && (weight_==0.0))
699  {
700  // synapse prepares to be picked up by the garbage collector.
701  // invalid value of -1.0 for psp_facilitation_ is used to indicate
702  // synapses to be deleted. The synapse will be removed next time
703  // when the garbage collector is invoked.
704  psp_facilitation_ = -1.0;
705  nest::synindex syn_id = nest::Connection<targetidentifierT>::get_syn_id();
706  ConnectionUpdateManager::instance()->trigger_garbage_collector(get_target(thread)->get_gid(),
707  e.get_sender_gid(), thread, syn_id );
708  return;
709  }
710 
711  // Make sure that the event is not a SynapseUpdateEvent.
712  if (e.get_rport() >= 0)
713  {
714  // Apply presynaptic spike
715  psp_facilitation_ += 1.0;
716  psp_depression_ += 1.0;
717 
718  if (weight_ > 0.0)
719  {
720  e.set_weight(weight_);
721 
722  e.set_delay(get_delay_steps());
723  e.set_receiver(*get_target(thread));
724  e.set_rport(get_rport());
725  e();
726  }
727  }
728 }

745 template <typename targetidentifierT>
746 void SynapticSamplingRewardGradientConnection<targetidentifierT>::update_synapse_state(long t_to,
747  long t_last_update,
748  TracingNode::const_iterator
749  &bap_trace,
750  TracingNode::const_iterator
751  &dopa_trace,
752  const CommonPropertiesType& cp)
753 {
754  if ((weight_ == 0.0) && not cp.simulate_retracted_synapses_)
755  {
756  // synapse is retracted. psps and eligibility traces are not going to be simulated.
757  return;
758  }
759 
760  assert(t_to >= t_last_update);
761  long steps = t_to - t_last_update;
762 
763  const double sc_psp = weight_ * cp.psp_scale_factor_;
764  const bool direct_gradient = cp.direct_gradient_rate_ > 0.0;
765  bool psp_active = (psp_facilitation_ != 0.0);
766 
767  while( steps )
768  {
769  // This loop will - considering every call - iterate through EVERY time step (in steps of resolution)
770 
771  // decay eligibility trace
772  eligibility_trace_ *= cp.eligibility_trace_update_;
773 
774  // decay gradient variable
775  reward_gradient_ *= cp.reward_gradient_update_;
776 
777  // update postsynaptic spike potential
778  if (psp_active)
779  {
780  psp_facilitation_ *= cp.psp_faciliation_update_;
781  psp_depression_ *= cp.psp_depression_update_;
782 
783  eligibility_trace_ += sc_psp * (psp_facilitation_ - psp_depression_) * (*bap_trace);
784 
785  if (psp_facilitation_ < cp.psp_cutoff_amplitude_)
786  {
787  psp_facilitation_ = 0.0;
788  psp_depression_ = 0.0;
789  psp_active = false;
790  }
791  }
792 
793  reward_gradient_ += (*dopa_trace) * eligibility_trace_;
794 
795  if (direct_gradient)
796  {
797  synaptic_parameter_ += (*dopa_trace) * cp.learning_rate_ *
798  cp.direct_gradient_rate_ * eligibility_trace_;
799  }
800 
801  ++bap_trace;
802  ++dopa_trace;
803  --steps;
804  }
805 }

815 template < typename targetidentifierT >
816 void SynapticSamplingRewardGradientConnection< targetidentifierT >::
817 update_synapic_parameter(nest::thread thread, const CommonPropertiesType& cp)
818 {
819  // update synaptic parameters
820  const double l_rate = cp.weight_update_interval_ * cp.learning_rate_;
821 
822  // compute prior
823  const double prior = prior_precision_ * (prior_mean_ - synaptic_parameter_);
824 
825  reward_gradient_ += cp.get_gradient_noise(thread);
826 
827  const double d_lik = std::max(-cp.max_param_change_,
828  std::min(cp.max_param_change_, cp.gradient_scale_ * reward_gradient_));
829 
830  const double d_param = l_rate * (prior + d_lik) + cp.get_d_wiener(thread);
831 
832  synaptic_parameter_ = std::max(cp.min_param_, std::min(cp.max_param_, synaptic_parameter_ + d_param));
833 }

843 template < typename targetidentifierT >
844 void SynapticSamplingRewardGradientConnection< targetidentifierT >::
845 update_synapic_weight(long time_step, const CommonPropertiesType& cp)
846 {
847  const bool synapse_is_active = (weight_ != 0.0) || (time_step==0);
848 
849  // update synaptic weight
850  if (synaptic_parameter_ >= 0.0)
851  {
852  weight_ = cp.weight_scale_ * std::exp(synaptic_parameter_ - cp.parameter_mapping_offset_);
853  }
854  else
855  {
856  weight_ = 0.0;
857  }
858 
859  if (synapse_is_active && not cp.simulate_retracted_synapses_ && (weight_ == 0.0))
860  {
861  // synapse is entering the retracted state, eligibility trace is reset.
862  psp_facilitation_ = 0.0;
863  psp_depression_ = 0.0;
864  eligibility_trace_ = 0.0;
865  reward_gradient_ = 0.0;
866  }
867 
868  logger()->record(time_step*cp.resolution_unit_, *this, recorder_port_);
869 }
