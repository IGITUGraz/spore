698 template <typename targetidentifierT>
699 void SynapticSamplingRewardGradientConnection<targetidentifierT>::send( nest::Event& e,
700  nest::thread thread,
701  double t_last_spike,
702  const CommonPropertiesType &cp )
703 {
704  assert(cp.resolution_unit_>0.0);
705 
706  const double t_to = e.get_stamp().get_ms(); // ConnectionUpdateManager::instance()->get_origin().get_ms();
707  double t_from = t_last_spike;
708 
709  if (t_to>t_last_spike)
710  {
711  // prepare the pointer to the target neuron. We can safely static_cast
712  // since the connection is checked when established.
713  TracingNode* target = static_cast<TracingNode*> (get_target(thread));
714 
715  TracingNode::const_iterator bap_trace = target->get_trace( nest::delay(t_from/cp.resolution_unit_), cp.bap_trace_id_ + get_rport() );
716  TracingNode::const_iterator dopa_trace = cp.reward_transmitter_->get_trace( nest::delay(t_from/cp.resolution_unit_), cp.dopa_trace_id_ );
717 
718  for ( double next_weight_time = t_weight_ + cp.weight_update_time_;
719  next_weight_time <= t_to;
720  next_weight_time += cp.weight_update_time_ )
721  {
722  update_synapse_state(next_weight_time, t_from, bap_trace, dopa_trace, cp);
723  update_synapic_weight(next_weight_time, thread, target, cp);
724  t_from = next_weight_time;
725  }
726 
727  if (t_to > t_from)
728  {
729  update_synapse_state(t_to, t_from, bap_trace, dopa_trace, cp);
730  }
731  }
732 
733  // Make sure that the event is not a SynapseUpdateEvent.
734  if (e.get_rport() >= 0)
735  {
736  // Apply presynaptic spike
737  psp_facilitation_ += 1.0;
738  psp_depression_ += 1.0;
739 
740  if (weight_ > 0.0)
741  {
742  e.set_weight(weight_);
743 
744  e.set_delay(get_delay_steps());
745  e.set_receiver(*get_target(thread));
746  e.set_rport(get_rport());
747  e();
748  }
749  }
750 }

766 template <typename targetidentifierT>
767 void SynapticSamplingRewardGradientConnection<targetidentifierT>::update_synapse_state( double t_to,
768  double t_last_update,
769  TracingNode::const_iterator &bap_trace,
770  TracingNode::const_iterator &dopa_trace,
771  const CommonPropertiesType& cp )
772 {
773  t_to -= cp.resolution_unit_/2.0; // exclude the last time step.
774 
775  for (double time = t_last_update; time < t_to; time += cp.resolution_unit_)
776  {
777  // This loop will - considering every call - iterate through EVERY time step (in steps of resolution)
778 
779  // decay eligibility trace
780  stdp_eligibility_trace_ *= cp.lambda_;
781 
782  // decay gradient variable
783  reward_gradient_ *= cp.gamma_;
784 
785  // update postsynaptic spike potential
786  if (psp_facilitation_ > 0.0)
787  {
788  psp_facilitation_ *= cp.psp_faciliation_update_;
789  psp_depression_ *= cp.psp_depression_update_;
790 
791  const double postsynaptic_potential = cp.psp_scale_factor_ * (psp_facilitation_ - psp_depression_);
792  stdp_eligibility_trace_ += weight_ * postsynaptic_potential * (*bap_trace);
793 
794  if (psp_facilitation_ < cp.epsilon_)
795  {
796  psp_facilitation_ = 0.0;
797  psp_depression_ = 0.0;
798  }
799  }
800 
801  reward_gradient_ += (*dopa_trace) * stdp_eligibility_trace_;
802 
803  if (cp.direct_gradient_rate_ > 0.0)
804  {
805  synaptic_parameter_ += (*dopa_trace) * cp.learning_rate_ * cp.direct_gradient_rate_*stdp_eligibility_trace_;
806  }
807 
808  ++bap_trace;
809  ++dopa_trace;
810  }
811 }

822 template <typename targetidentifierT>
823 void SynapticSamplingRewardGradientConnection<targetidentifierT>::update_synapic_weight(double time,
824  nest::thread thread,
825  TracingNode *target,
826  const CommonPropertiesType& cp)
827 {
828  // update synaptic parameters
829  const double l_rate = cp.weight_update_time_ * cp.learning_rate_;
830 
831  if ((time >= cp.consolidation_time_) && (t_weight_ < cp.consolidation_time_))
832  {
833  if (cp.drand(thread) <= cp.consolidation_probability_)
834  {
835  prior_mean_ = synaptic_parameter_;
836  prior_std_ = cp.consolidation_prior_std_;
837  t_consolidate_ = -1.0;
838 
839  if (cp.verbose_)
840  {
841  std::cout << "time: " << time
842  << ", consolidating synapse to parameter value :" << synaptic_parameter_ << std::endl;
843  }
844  }
845  else if (cp.verbose_)
846  {
847  std::cout << "time: " << time << ", consolidation skipped." << std::endl;
848  }
849  }
850 
851  if ((cp.consolidation_interval_ > 0.0) && (t_consolidate_ >= 0.0))
852  {
853  if (synaptic_parameter_ > cp.consolidation_threshold_)
854  {
855  if (t_consolidate_ > cp.consolidation_interval_)
856  {
857  prior_mean_ = synaptic_parameter_;
858  prior_std_ = cp.consolidation_prior_std_;
859  t_consolidate_ = -1.0;
860  }
861  else
862  {
863  t_consolidate_ += cp.weight_update_time_;
864  }
865  }
866  else
867  {
868  t_consolidate_ = 0.0;
869  }
870  }
871 
872  // compute prior
873  double prior = (1.0 / pow(prior_std_, 2)) * (prior_mean_ - synaptic_parameter_);
874 
875  reward_gradient_ += cp.get_gradient_noise(thread);
876
