573 template <typename targetidentifierT>
574 void SynapticSamplingRewardGradientConnection<targetidentifierT>::send(nest::Event& e,
575  nest::thread thread,
576  double t_last_spike,
577  const CommonPropertiesType &cp)
578 {
579  if (is_degenerated())
580  {
581  // synapse is waiting for the garbage collector.
582  return;
583  }
584 
585  assert(cp.resolution_unit_ > 0.0);
586 
587  const long s_to = std::floor( e.get_stamp().get_ms() / cp.resolution_unit_ );
588  long s_from = std::floor( t_last_spike / cp.resolution_unit_ );
589 
590  if (s_to > s_from)
591  {
592  if (s_from == 0)
593  {
594  update_synapic_weight(0, cp);
595  }
596  // prepare the pointer to the target neuron. We can safely static_cast
597  // since the connection is checked when established.
598  TracingNode* target = static_cast<TracingNode*> (get_target(thread));
599 
600  TracingNode::const_iterator bap_trace =
601  target->get_trace(s_from, cp.bap_trace_id_);
602 
603  TracingNode::const_iterator dopa_trace =
604  cp.reward_transmitter_->get_trace(s_from, cp.dopa_trace_id_);
605 
606  const double t_last_weight_update = std::floor(t_last_spike / cp.weight_update_time_) * cp.weight_update_time_;
607  const long s_last_update = std::floor( t_last_weight_update/cp.resolution_unit_ );
608 
609  for (long next_weight_step = s_last_update + cp.weight_update_steps_;
610  next_weight_step <= s_to;
611  next_weight_step += cp.weight_update_steps_)
612  {
613  update_synapse_state(next_weight_step, s_from, bap_trace, dopa_trace, cp);
614  update_synapic_parameter(thread, cp);
615  update_synapic_weight(next_weight_step, cp);
616  s_from = next_weight_step;
617  }
618 
619  if (s_to > s_from)
620  {
621  update_synapse_state(s_to, s_from, bap_trace, dopa_trace, cp);
622  }
623  }
624 
625  if (cp.delete_retracted_synapses_ && (weight_==0.0))
626  {
627  // synapse prepares to be picked up by the garbage collector.
628  // invalid value of -1.0 for psp_facilitation_ is used to indicate
629  // synapses to be deleted. The synapse will be removed next time
630  // when the garbage collector is invoked.
631  psp_facilitation_ = -1.0;
632  nest::synindex syn_id = nest::Connection<targetidentifierT>::get_syn_id();
633  ConnectionUpdateManager::instance()->trigger_garbage_collector(get_target(thread)->get_gid(),
634  e.get_sender_gid(), thread, syn_id );
635  return;
636  }
637 
638  // Make sure that the event is not a SynapseUpdateEvent.
639  if (e.get_rport() >= 0)
640  {
641  // Apply presynaptic spike
642  psp_facilitation_ += 1.0;
643  psp_depression_ += 1.0;
644 
645  if (weight_ > 0.0)
646  {
647  e.set_weight(weight_);
648 
649  e.set_delay(get_delay_steps());
650  e.set_receiver(*get_target(thread));
651  e.set_rport(get_rport());
652  e();
653  }
654  }
655 }

670 template <typename targetidentifierT>
671 void SynapticSamplingRewardGradientConnection<targetidentifierT>::update_synapse_state(long t_to,
672  long t_last_update,
673  TracingNode::const_iterator
674  &bap_trace,
675  TracingNode::const_iterator
676  &dopa_trace,
677  const CommonPropertiesType& cp)
678 {
679  if ((weight_ == 0.0) && not cp.simulate_retracted_synapses_)
680  {
681  // synapse is retracted. psps and eligibility traces are not going to be simulated.
682  return;
683  }
684 
685  assert(t_to >= t_last_update);
686  long steps = t_to - t_last_update;
687 
688  const double sc_psp = weight_ * cp.psp_scale_factor_;
689  const bool direct_gradient = cp.direct_gradient_rate_ > 0.0;
690  bool psp_active = (psp_facilitation_ != 0.0);
691 
692  while( steps )
693  {
694  // This loop will - considering every call - iterate through EVERY time step (in steps of resolution)
695 
696  // decay eligibility trace
697  stdp_eligibility_trace_ *= cp.lambda_;
698 
699  // decay gradient variable
700  reward_gradient_ *= cp.gamma_;
701 
702  // update postsynaptic spike potential
703  if (psp_active)
704  {
705  psp_facilitation_ *= cp.psp_faciliation_update_;
706  psp_depression_ *= cp.psp_depression_update_;
707 
708  stdp_eligibility_trace_ += sc_psp * (psp_facilitation_ - psp_depression_) * (*bap_trace);
709 
710  if (psp_facilitation_ < cp.epsilon_)
711  {
712  psp_facilitation_ = 0.0;
713  psp_depression_ = 0.0;
714  psp_active = false;
715  }
716  }
717 
718  reward_gradient_ += (*dopa_trace) * stdp_eligibility_trace_;
719 
720  if (direct_gradient)
721  {
722  synaptic_parameter_ += (*dopa_trace) * cp.learning_rate_ *
723  cp.direct_gradient_rate_ * stdp_eligibility_trace_;
724  }
725 
726  ++bap_trace;
727  ++dopa_trace;
728  --steps;
729  }
730 }

738 template < typename targetidentifierT >
739 void SynapticSamplingRewardGradientConnection< targetidentifierT >::
740 update_synapic_parameter(nest::thread thread, const CommonPropertiesType& cp)
741 {
742  // update synaptic parameters
743  const double l_rate = cp.weight_update_time_ * cp.learning_rate_;
744 
745  // compute prior
746  const double prior = prior_scale_factor_ * (prior_mean_ - synaptic_parameter_);
747 
748  reward_gradient_ += cp.get_gradient_noise(thread);
749 
750  const double d_lik = std::max(-cp.max_param_change_,
751  std::min(cp.max_param_change_, cp.gradient_scale_ * reward_gradient_));
752 
753  const double d_param = l_rate * (prior + d_lik) + cp.get_d_wiener(thread);
754 
755  synaptic_parameter_ = std::max(cp.min_param_, std::min(cp.max_param_, synaptic_parameter_ + d_param));
756 }

764 template < typename targetidentifierT >
765 void SynapticSamplingRewardGradientConnection< targetidentifierT >::
766 update_synapic_weight(long time_step, const CommonPropertiesType& cp)
767 {
768  const bool synapse_is_active = (weight_ != 0.0) || (time_step==0);
769 
770  // update synaptic weight
771  if (synaptic_parameter_ >= 0.0)
772  {
773  weight_ = std::exp(synaptic_parameter_ - cp.parameter_mapping_offset_);
774  }
775  else
776  {
777  weight_ = 0.0;
778  }
779 
780  if (synapse_is_active && not cp.simulate_retracted_synapses_ && (weight_ == 0.0))
781  {
782  // synapse is entering the retracted state, eligibility trace is reset.
783  psp_facilitation_ = 0.0;
784  psp_depression_ = 0.0;
785  stdp_eligibility_trace_ = 0.0;
786  reward_gradient_ = 0.0;
787  }
788 
789  logger()->record(time_step*cp.resolution_unit_, *this, recorder_port_);
790 }
