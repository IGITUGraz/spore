265 template <typename targetidentifierT>
266 void SporeTestConnection<targetidentifierT>::send(nest::Event& e,
267  nest::thread thread,
268  double t_last_spike,
269  const CommonPropertiesType &cp)
270 {
271  const double t_to = e.get_stamp().get_ms(); // ConnectionUpdateManager::instance()->get_origin().get_ms();
272  double t_from = t_last_spike;
273 
274  assert(cp.resolution_unit_ > 0.0);
275 
276  if (t_to > t_last_spike)
277  {
278  // prepare the pointer to the target neuron. We can safely static_cast
279  // since the connection is checked when established.
280  TracingNode* target = static_cast<TracingNode*> (get_target(thread));
281 
282  TracingNode::const_iterator bap_trace = target->get_trace(nest::delay(t_from / cp.resolution_unit_), cp.bap_trace_id_ + get_rport());
283 
284  for (double next_weight_time = t_weight_ + cp.weight_update_time_;
285  next_weight_time <= t_to;
286  next_weight_time += cp.weight_update_time_)
287  {
288  update_synapse_state(next_weight_time, t_from, bap_trace, cp);
289  update_synapic_weight(next_weight_time, thread, target, cp);
290  t_from = next_weight_time;
291  }
292 
293  if (t_to > t_from)
294  {
295  update_synapse_state(t_to, t_from, bap_trace, cp);
296  }
297  }
298 
299  if (e.get_rport() >= 0)
300  {
301  if (weight_ > 0.0)
302  {
303  e.set_weight(weight_);
304 
305  e.set_delay(get_delay_steps());
306  e.set_receiver(*get_target(thread));
307  e.set_rport(get_rport());
308  e();
309  }
310  }
311 }

325 template <typename targetidentifierT>
326 void SporeTestConnection<targetidentifierT>::update_synapse_state(double t_to,
327  double t_last_update,
328  TracingNode::const_iterator &bap_trace,
329  const CommonPropertiesType& cp)
330 {
331  t_to -= cp.resolution_unit_ / 2.0; // exclude the last time step.
332 
333  for (double time = t_last_update; time < t_to; time += cp.resolution_unit_)
334  {
335  assert(time >= ConnectionUpdateManager::instance()->get_horizon().get_ms() &&
336  time < ConnectionUpdateManager::instance()->get_origin().get_ms());
337 
338  recorder_times_.push_back(time);
339  recorder_values_.push_back(*bap_trace);
340  ++bap_trace;
341  }
342 }
