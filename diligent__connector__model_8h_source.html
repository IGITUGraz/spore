248 template < typename ConnectionT >
249 nest::ConnectorBase* DiligentConnectorModel< ConnectionT >::get_hom_connector(nest::ConnectorBase* conn,
250  nest::synindex syn_id)
251 {
252  if (!conn)
253  {
254  return 0;
255  }
256  else if (conn->homogeneous_model())
257  {
258  if (conn->get_syn_id() == syn_id)
259  return conn;
260  }
261  else
262  {
263  // connector is heterogeneous - go through all entries and search for correct syn_id
264  nest::HetConnector* hc = static_cast<nest::HetConnector*> (conn);
265  for (size_t i = 0; i < hc->size(); i++)
266  {
267  // need to cast to vector_like to access syn_id
268  if ((*hc)[ i ]->get_syn_id() == syn_id) // find entry for this type
269  {
270  return ( *hc)[ i ];
271  }
272  }
273  }
274 
275  return 0;
276 }
