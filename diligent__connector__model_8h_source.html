272 template < typename ConnectionT >
273 nest::ConnectorBase* DiligentConnectorModel< ConnectionT >::cleanup_delete_connection(nest::Node& tgt,
274  size_t target_thread,
275  nest::ConnectorBase* conn,
276  nest::synindex syn_id)
277 {
278  using nest::HetConnector;
279  using nest::vector_like;
280  using nest::pack_pointer;
281  using nest::kernel;
282 
283  assert(conn != 0); // we should not delete not existing synapses
284  bool found = false;
285  vector_like< ConnectionT >* vc;
286 
287  bool b_has_primary = has_primary(conn);
288  bool b_has_secondary = has_secondary(conn);
289 
290  conn = validate_pointer(conn);
291  // from here on we can use conn as a valid pointer
292 
293  if (conn->homogeneous_model())
294  {
295  assert(conn->get_syn_id() == syn_id);
296  vc = static_cast<vector_like< ConnectionT >*> (conn);
297  // delete the first Connection corresponding to the target
298  for (size_t i = 0; i < vc->size(); i++)
299  {
300  ConnectionT* connection = &vc->at(i);
301 
302  // only remove synapse if marked for deletion.
303  if ((connection->get_target(target_thread)->get_gid() == tgt.get_gid()) && connection->is_degenerated())
304  {
305  if (vc->get_num_connections() > 1)
306  conn = &vc->erase(i);
307  else
308  {
309  delete vc;
310  conn = 0;
311  }
312  bool is_primary = DiligentConnectorModel< ConnectionT >::is_primary_;
313  if (conn != 0)
314  conn = pack_pointer(conn, is_primary, !is_primary);
315  found = true;
316  break;
317  }
318  }
319  }
320  else
321  {
322  // heterogeneous case
323  // go through all entries and search for correct syn_id
324  // if not found create new entry for this syn_id
325  HetConnector* hc = static_cast<HetConnector*> (conn);
326 
327  for (size_t i = 0; i < hc->size() && !found; i++)
328  {
329  // need to cast to vector_like to access syn_id there is already an entry
330  // for this type
331  if ((*hc)[ i ]->get_syn_id() == syn_id)
332  {
333  // here we know that the type is vector_like<connectionT>, because
334  // syn_id agrees so we can safely static cast
335  vector_like< ConnectionT >* vc =
336  static_cast<vector_like< ConnectionT >*> ((*hc)[ i ]);
337  // Find and delete the first Connection corresponding to the target
338  for (size_t j = 0; j < vc->size(); j++)
339  {
340  ConnectionT* connection = &vc->at(j);
341 
342  if ((connection->get_target(target_thread)->get_gid() == tgt.get_gid()) &&
343  connection->is_degenerated()) // only remove synapse if marked for deletion.
344  {
345  // Get rid of the ConnectionBase for this type of synapse if there
346  // is only this element left
347  if (vc->size() == 1)
348  {
349  (*hc).erase((*hc).begin() + i);
350  // Test if the homogeneous vector of connections went back to only
351  // 1 type of synapse... then go back to the simple vector_like
352  // case.
353  if (hc->size() == 1)
354  {
355  conn = (*hc)[ 0 ];
356  const bool is_primary =
357  kernel()
358  .model_manager.get_synapse_prototype(conn->get_syn_id())
359  .is_primary();
360  conn = pack_pointer(conn, is_primary, not is_primary);
361  }
362  else
363  {
364  conn = pack_pointer(hc, b_has_primary, b_has_secondary);
365  }
366  } // Otherwise, just remove the desired connection
367  else
368  {
369  (*hc)[ i ] = &vc->erase(j);
370  conn = pack_pointer(hc, b_has_primary, b_has_secondary);
371  }
372  found = true;
373  break;
374  }
375  }
376  }
377  }
378  }
379 
380  if (found)
381  {
382  return conn;
383  }
384  else
385  {
386  return nest::GenericConnectorModel< ConnectionT >::delete_connection(tgt, target_thread, conn, syn_id);
387  }
388 }

402 template < typename ConnectionT >
403 nest::ConnectorBase* DiligentConnectorModel< ConnectionT >::get_hom_connector(nest::ConnectorBase* conn,
404  nest::synindex syn_id)
405 {
406  if (!conn)
407  {
408  return 0;
409  }
410  else if (conn->homogeneous_model())
411  {
412  if (conn->get_syn_id() == syn_id)
413  return conn;
414  }
415  else
416  {
417  // connector is heterogeneous - go through all entries and search for correct syn_id
418  nest::HetConnector* hc = static_cast<nest::HetConnector*> (conn);
419  for (size_t i = 0; i < hc->size(); i++)
420  {
421  // need to cast to vector_like to access syn_id
422  if ((*hc)[ i ]->get_syn_id() == syn_id) // find entry for this type
423  {
424  return ( *hc)[ i ];
425  }
426  }
427  }
428 
429  return 0;
430 }
