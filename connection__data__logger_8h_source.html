134 template<typename ConnectionType>
135 void ConnectionDataLogger<ConnectionType>::record(double time_step,
136  ConnectionType const &host,
137  recorder_port port)
138 {
139  if (port == nest::invalid_index)
140  return;
141 
142  assert(port < recorder_data_.size());
143  ConnectionDataLoggerBase::RecorderData &recorder = *recorder_data_[port];
144 
145  if (recorder.interval_ == 0)
146  return;
147 
148  if (recorder.recorder_times_.empty() || (recorder.recorder_times_.back() + recorder.interval_ <= time_step))
149  {
150  recorder.recorder_times_.push_back(time_step);
151 
152  for (size_t i = 0; i < recorder.recorder_values_.size(); i++)
153  {
154  recorder.recorder_values_[i].push_back(((host).*(data_access_fct_[i]))());
155  }
156  }
157 }
